Non-Linear Data Structures

Hierarchical and interconnected data organization techniques

Trees (Binary Trees, BST, AVL)

Definition: Trees are hierarchical data structures consisting of nodes connected by edges. Binary trees have at most two children per node, BSTs maintain ordering, and AVL trees are self-balancing BSTs.

Usage: Used for hierarchical data representation, efficient searching (BST), and maintaining balanced trees for optimal performance (AVL).

Real-life Analogy: Like a family tree or organizational chart. BST is like a dictionary where words are arranged alphabetically, and AVL is like a balanced scale ensuring efficiency.

Binary Search Tree Implementation

class Node {
public:
    int data;
    Node* left;
    Node* right;
    Node(int val) : data(val), left(nullptr), right(nullptr) {}
};

class BST {
private:
    Node* root;
    
    Node* insertRec(Node* node, int val) {
        if (!node) return new Node(val);
        if (val < node->data) 
            node->left = insertRec(node->left, val);
        else if (val > node->data)
            node->right = insertRec(node->right, val);
        return node;
    }
    
public:
    BST() : root(nullptr) {}
    
    void insert(int val) {
        root = insertRec(root, val);
    }
    
    bool search(int val) {
        Node* curr = root;
        while (curr) {
            if (val == curr->data) return true;
            if (val < curr->data) curr = curr->left;
            else curr = curr->right;
        }
        return false;
    }
};

#include <iostream>
#include <queue>
using namespace std;

int main() {
    // Max-heap
    priority_queue<int> max_heap;
    max_heap.push(30);
    max_heap.push(100);
    max_heap.push(25);
    max_heap.push(40);
    
    cout << "Max element: " << max_heap.top() << endl;
    max_heap.pop();
    cout << "New max: " << max_heap.top() << endl;

    priority_queue<int, vector<int>, greater<int>> min_heap;

    min_heap.push(100);
    min_heap.push(25);
    min_heap.push(40);
    
    cout << "Min element: " << min_heap.top() << endl;
    min_heap.pop();
    cout << "New min: " << min_heap.top() << endl;

}

Graphs (Adjacency, Traversal, Algorithms)

Definition: Graphs consist of vertices (nodes) connected by edges. Can be directed/undirected and weighted/unweighted. Common representations: adjacency list/matrix.

Usage: Modeling networks (social, transportation), pathfinding algorithms, dependency resolution, and recommendation systems.

Real-life Analogy: Like a city map with intersections (vertices) and roads (edges), or a social network with people (vertices) and friendships (edges).

Breadth-First Search (BFS)

void BFS(vector<list<int>>& graph, int start) {
    vector<bool> visited(graph.size(), false);
    queue<int> q;
    q.push(start);
    visited[start] = true;
    
    while (!q.empty()) {
        int node = q.front();
        q.pop();
        cout << node << " ";
        
        for (int neighbor : graph[node]) {
            if (!visited[neighbor]) {
                visited[neighbor] = true;
                q.push(neighbor);
            }
        }
    }
}

Hash Tables (Collision Handling, Applications)

Definition: Hash tables store key-value pairs using a hash function to compute an index into an array of buckets. Collisions occur when multiple keys map to the same bucket.

Usage: Fast data retrieval, database indexing, caches, and associative arrays.

Real-life Analogy: Like a library catalog system where books (values) are located using their call numbers (keys), with multiple books sometimes sharing the same location (collision).

Collision Handling Methods

// Chaining: Linked list at each bucket
class HashTable {
private:
    vector<list<pair<int, string>>> table;
    int size;
    
    int hash(int key) { return key % size; }
    
public:
    HashTable(int s) : size(s) { table.resize(size); }
    
    void insert(int key, string value) {
        int index = hash(key);
        table[index].push_back({key, value});
    }
    
    string search(int key) {
        int index = hash(key);
        for (auto& p : table[index]) {
            if (p.first == key) return p.second;
        }
        return "Not Found";
    }
};

// Open Addressing: Linear probing
class HashTableOA {
private:
    vector<pair<int, string>> table;
    int size;
    
    int hash(int key) { return key % size; }
    
public:
    HashTableOA(int s) : size(s) { table.resize(size, {-1, ""}); }
    
    void insert(int key, string value) {
        int index = hash(key);
        while (table[index].first != -1 && table[index].first != key) {
            index = (index + 1) % size;
        }
        table[index] = {key, value};
    }
    
    string search(int key) {
        int index = hash(key);
        int start = index;
        while (table[index].first != -1) {
            if (table[index].first == key) return table[index].second;
            index = (index + 1) % size;
            if (index == start) break;
        }
        return "Not Found";
    }
};

Tries (Prefix Trees, Applications)

Usage: Autocomplete systems, spell checkers, IP routing, and dictionary implementations.

Trie Implementation

class TrieNode {
public:
    unordered_map<char, TrieNode*> children;
    bool isEndOfWord;
    
    TrieNode() : isEndOfWord(false) {}
};

class Trie {
private:
    TrieNode* root;
    
public:
    Trie() {
        root = new TrieNode();
    }
    
    void insert(string word) {
        TrieNode* node = root;
        for (char c : word) {
            if (node->children.find(c) == node->children.end()) {
                node->children[c] = new TrieNode();
            }
            node = node->children[c];
        }
        node->isEndOfWord = true;
    }
    
    bool search(string word) {
        TrieNode* node = root;
        for (char c : word) {
            if (node->children.find(c) == node->children.end()) {
                return false;
            }
            node = node->children[c];
        }
        return node->isEndOfWord;
    }
};

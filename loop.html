16. Check if a number is an Armstrong number 
int original = num, sum = 0;

21. Find the N-th Fibonacci number using iteration 
int a = 0, b = 1, fib;
if(N == 1) fib = a;
else if(N == 2) fib = b;
else {
    for(int i = 3; i <= N; i++) {
        fib = a + b;
        a = b;
        b = fib;
    }
}
cout << fib;

Segment Trees

Definition: Efficient data structure for range queries and point updates.

Usage: Range minimum/maximum queries, range sum queries, and other associative operations on arrays.

Real-life Analogy: Like a multi-level filing system where each folder contains information about a specific range of documents, allowing you to quickly find information about any section.

Implementation (Range Sum)

class SegmentTree {
    vector<int> tree;
    int n;
    
    void build(vector<int>& nums, int node, int left, int right) {
        if (left == right) {
            tree[node] = nums[left];
            return;
        }
        int mid = left + (right - left) / 2;
        build(nums, 2*node+1, left, mid);
        build(nums, 2*node+2, mid+1, right);
        tree[node] = tree[2*node+1] + tree[2*node+2];
    }
    
public:
    SegmentTree(vector<int>& nums) {
        n = nums.size();
        tree.resize(4*n);
        build(nums, 0, 0, n-1);
    }
    
    void update(int index, int val, int node=0, int left=0, int right=-1) {
        if (right == -1) right = n-1;
        if (left == right) {
            tree[node] = val;
            return;
        }
        int mid = left + (right - left) / 2;
        if (index <= mid) update(index, val, 2*node+1, left, mid);

        tree[node] = tree[2*node+1] + tree[2*node+2];

    int query(int ql, int qr, int node=0, int left=0, int right=-1) {
        if (right == -1) right = n-1;
        if (qr < left || ql > right) return 0;

Construction O(n)

Point Update O(log n)
Variations: Can be adapted for min/max, gcd, or other associative operations

Fenwick Trees (BIT)

Definition: Efficient data structure for point updates and prefix sum queries.

Usage: Calculating prefix sums, range sum queries, and solving inversion count problems.

Real-life Analogy: Like a set of Russian dolls where each doll contains the sum of values inside the smaller dolls it contains.

Implementation

class FenwickTree {
    vector<int> bit;
    
public:
    FenwickTree(int size) : bit(size + 1) {}
    
    void update(int index, int delta) {
        for (; index < bit.size(); index += index & -index)
            bit[index] += delta;
    }
    
    int query(int index) {
        int sum = 0;
        for (; index > 0; index -= index & -index)
            sum += bit[index];
        return sum;
    }
    
    int rangeQuery(int l, int r) {
        return query(r) - query(l - 1);
    }
};

Trie (Prefix Tree)

Definition: Tree-like data structure for efficient string storage and prefix searches.

Usage: Autocomplete systems, spell checkers, and IP routing tables.

Real-life Analogy: Like a dictionary index where words are organized by their prefixes, allowing you to quickly find all words starting with a specific prefix.

Implementation

class TrieNode {
public:
    unordered_map<char, TrieNode*> children;
    bool is_word = false;
};

class Trie {
    TrieNode* root;
    
public:
    Trie() : root(new TrieNode()) {}
    
    void insert(string word) {
        TrieNode* node = root;
        for (char c : word) {
            if (!node->children.count(c))
                node->children[c] = new TrieNode();
            node = node->children[c];
        }
        node->is_word = true;
    }
    
    bool search(string word) {
        TrieNode* node = root;
        for (char c : word) {
            if (!node->children.count(c)) return false;
            node = node->children[c];
        }
        return node->is_word;
    }
    
    bool startsWith(string prefix) {
        TrieNode* node = root;
        for (char c : prefix) {
            if (!node->children.count(c)) return false;
            node = node->children[c];
        }
        return true;
    }
};

Applications

  • Autocomplete systems
  • Spell checking
  • IP routing (longest prefix matching)
  • Word games (Boggle, Scrabble)

Disjoint Set Union (DSU)

Definition: Data structure that tracks a set of elements partitioned into disjoint subsets.

Usage: Kruskal's algorithm, connected components in graphs, and percolation problems.

Real-life Analogy: Like a network of friends where each person knows who their representative is, allowing efficient grouping and merging of friend circles.

Implementation with Path Compression

class DSU {
    vector<int> parent, rank;
    
public:
    DSU(int n) : parent(n), rank(n, 1) {
        iota(parent.begin(), parent.end(), 0);
    }
    
    int find(int u) {
        return parent[u] == u ? u : parent[u] = find(parent[u]);
    }
    
    void unite(int u, int v) {
        u = find(u), v = find(v);
        if (u == v) return;
        if (rank[u] < rank[v]) swap(u, v);
        parent[v] = u;
        if (rank[u] == rank[v]) rank[u]++;
    }
    
    bool connected(int u, int v) {
        return find(u) == find(v);
    }
};

Complexity

Operation Time
Find O(Œ±(n))
Union O(Œ±(n))
Note: Œ±(n) is the inverse Ackermann function (effectively constant)

Suffix Automaton

Definition: Compact automaton representing all suffixes of a string.

Usage: String pattern matching, longest common substring, and substring counting.

Real-life Analogy: Like a specialized factory assembly line that efficiently produces and recognizes all possible suffixes of a product ID.

Implementation

struct State {
    int len, link;
    map<char, int> next;
};

class SuffixAutomaton {
    vector<State> st;
    int last;
    
public:
    SuffixAutomaton() {
        st.emplace_back();
        st[0].len = 0;
        st[0].link = -1;
        last = 0;
    }
    
    void extend(char c) {
        int p = last;
        st.emplace_back();
        int curr = st.size() - 1;
        st[curr].len = st[p].len + 1;
        
        while (p >= 0 && !st[p].next.count(c)) {
            st[p].next[c] = curr;
            p = st[p].link;
        }
        
        if (p == -1) {
            st[curr].link = 0;
        } else {
            int q = st[p].next[c];
            if (st[p].len + 1 == st[q].len) {
                st[curr].link = q;
            } else {
                st.emplace_back();
                int clone = st.size() - 1;
                st[clone].len = st[p].len + 1;
                st[clone].next = st[q].next;
                st[clone].link = st[q].link;
                
                while (p >= 0 && st[p].next[c] == q) {
                    st[p].next[c] = clone;
                    p = st[p].link;
                }
                st[q].link = st[curr].link = clone;
            }
        }
        last = curr;
    }
};

Applications

  • Longest common substring of multiple strings
  • Substring search with wildcards
  • Counting distinct substrings
  • Finding all occurrences of patterns

Heavy-Light Decomposition

Definition: Technique for decomposing trees into chains to support efficient path queries.

Usage: Path queries in trees, LCA computation, and tree path aggregation.

Real-life Analogy: Like organizing a complex transportation network with express lanes (heavy paths) for faster travel between major hubs.

Core Concepts

class HLD {
    vector<int> parent, depth, heavy, head, pos;
    int cur_pos;
    SegmentTree seg_tree;
    
    int dfs(int u, vector<vector<int>>& adj) {
        int size = 1, max_c_size = 0;
        for (int v : adj[u]) {
            if (v != parent[u]) {
                parent[v] = u;
                depth[v] = depth[u] + 1;
                int c_size = dfs(v, adj);
                size += c_size;
                if (c_size > max_c_size) {
                    max_c_size = c_size;
                    heavy[u] = v;
                }
            }
        }
        return size;
    }
    
    void decompose(int u, int h, vector<vector<int>>& adj) {
        head[u] = h;
        pos[u] = cur_pos++;
        if (heavy[u] != -1)
            decompose(heavy[u], h, adj);
        for (int v : adj[u]) {
            if (v != parent[u] && v != heavy[u])
                decompose(v, v, adj);
        }
    }
    
public:
    HLD(vector<vector<int>>& adj) {
        int n = adj.size();
        parent = vector<int>(n, -1);
        depth = vector<int>(n);
        heavy = vector<int>(n, -1);
        head = vector<int>(n);
        pos = vector<int>(n);
        cur_pos = 0;
        
        dfs(0, adj);
        decompose(0, 0, adj);
    }
    
    int query(int u, int v) {
        int res = 0;
        for (; head[u] != head[v]; v = parent[head[v]]) {
            if (depth[head[u]] > depth[head[v]]) swap(u, v);
            res = max(res, seg_tree.query(pos[head[v]], pos[v]));
        }
        if (depth[u] > depth[v]) swap(u, v);
        res = max(res, seg_tree.query(pos[u], pos[v]));
        return res;
    }
};

Use Cases

  • Path queries/max in trees
  • Lowest common ancestor (LCA)
  • Tree path aggregation
  • Dynamic tree problems
